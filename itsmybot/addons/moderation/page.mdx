## Commands
<Properties>
  <Property name="/logs <user>">
    Show the logs of a user.
  </Property>
  <Property name="/clear <amount> [channel] [user]">
    Clear messages in a channel.
  </Property>
  <Property name="/warn <user> <reason> [silent]">
    Warn a user.
  </Property>
  <Property name="/tempban <user> <duration> [reason]">
    Temporarily ban a user.
  </Property>
</Properties>
